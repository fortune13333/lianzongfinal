#### **生成代码 (Mermaid)**

```mermaid
%%{init: {'theme': 'neutral', 'fontFamily': 'sans-serif'}}%%
graph TD
    Start(接收用户按键 <fa:fa-keyboard>) --> IsEnter{是否为<br>回车键?};
    IsEnter -- 否 --> Accumulate["累加到行缓冲区<br>并实时转发给设备"];
    Accumulate --> Start;
    
    IsEnter -- 是 --> Normalize["标准化命令<br>(去首尾空格、转小写、压缩内部空格)"];
    Normalize --> IsForbidden{命令是否在<br>禁止列表中?};
    
    IsForbidden -- "否 (合规)" --> Execute["发送换行符<br>执行命令 <fa:fa-check-circle>"];
    Execute --> ClearBuffer[清空行缓冲区];
    
    IsForbidden -- "是 (违规)" --> Intercept;
    subgraph Intercept["<fa:fa-ban> <b>拦截操作</b>"]
        direction TB
        SendCtrlU["向设备发送<br><b>行清除信号 (CTRL+U)</b>"]
        SendWarning[向前端发送<br>策略违规警告]
    end
    Intercept --> ClearBuffer;
    
    ClearBuffer --> Start;

    linkStyle default stroke-width:2px
    style Start,IsEnter,Accumulate,Normalize,IsForbidden,ClearBuffer,SendCtrlU,SendWarning stroke-width:2px
    style Execute fill:#dcfce7,stroke:#22c55e,stroke-width:2px
    style Intercept fill:#fef2f2,stroke:#ef4444,stroke-width:2px
```
